$(function() {
   //mask
   $('.amount').mask('#,###.##', {reverse : true});
   
   //function that will get the total amount by each class
   var total_amount =  function() {
      var sum = 0;
      
      $('.amount').each(function(){
         var num = $(this).val().replace(',','');
         
         if(num != 0) {
            sum += parseFloat(num);
         }
      });
      
      $('#total_amount').val(sum.toFixed(2));
   }
   
   //keyup handler
   $('.amount').keyup(function(){
      total_amount();
   });
   
});
